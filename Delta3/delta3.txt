<script>
  function left() {
            var element = document.getElementById("image1");
            element.style.left = parseInt(element.style.left) - 1 + 'px';
localStorage.lastx = parseInt(element.style.left) ;
            }

            function right() {
            var element = document.getElementById("image1");
            element.style.left = parseInt(element.style.left) + 1 + 'px';
localStorage.lastx = parseInt(element.style.left) ;

            }

            function up() {
            var element = document.getElementById("image1");
            element.style.top = parseInt(element.style.top) - 1 + 'px';
localStorage.lasty = parseInt(element.style.top) ;
            }

            function down() {
            var element = document.getElementById("image1");
            element.style.top = parseInt(element.style.top) + 1 + 'px';
localStorage.lasty = parseInt(element.style.top) ;
            }

            function moveSelection(evt) {
 var element = document.getElementById("image1");
localStorage.lastx = parseInt(element.style.left) ;
localStorage.lasty = parseInt(element.style.top) ;
                switch (evt.keyCode) {
                    case 37:
                    left();
                    break;
                    case 39:
                    right();
                    break;
                    case 38:
                    up();
                    break;
                    case 40:
                    down();
                    break;
                    }
                };
   
        function docReady()
        {  

document.getElementById("image1").style.left=Number(localStorage.lastx);
document.getElementById("image1").style.top=Number(localStorage.lasty);


      
          window.addEventListener('keydown', moveSelection);
        }
        

</script>


  </head>

  <body onload="docReady()" onkeydown="" onkeyup="" >
